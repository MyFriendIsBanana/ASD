def z_function(s):
    n = len(s)
    z = [0] * n
    l, r = 0, 0
    for i in range(1, n):
        if i <= r:
            z[i] = min(r - i + 1, z[i - l])
        while i + z[i] < n and s[z[i]] == s[i + z[i]]:
            z[i] += 1
        if i + z[i] - 1 > r:
            l, r = i, i + z[i] - 1
    return z

import matplotlib.pyplot as plt
import time
import random
import string

#   
def random_string(length):
    return ''.join(random.choices(string.ascii_lowercase, k=length))

#   (RLE)
def compress_string(s):
    compressed = ""
    count = 1
    for i in range(1, len(s)):
        if s[i] == s[i-1]:
            count += 1
        else:
            compressed += s[i-1] + str(count)
            count = 1
    compressed += s[-1] + str(count)
    return compressed

#   
def naive_search(text, pattern):
    positions = []
    for i in range(len(text) - len(pattern) + 1):
        match = True
        for j in range(len(pattern)):
            if text[i+j] != pattern[j]:
                match = False
                break
        if match:
            positions.append(i)
    return positions

# Z-
def z_function(s):
    n = len(s)
    z = [0]*n
    l = r = 0
    for i in range(1, n):
        if i <= r:
            z[i] = min(r - i + 1, z[i - l])
        while i+z[i] < n and s[z[i]] == s[i + z[i]]:
            z[i] += 1
        if i + z[i] - 1 > r:
            l, r = i, i + z[i] - 1
    return z

#   
sizes = list(range(100, 1100, 100))
compress_times = []
naive_times = []
zfunc_times = []

for size in sizes:
    s = random_string(size)
    pattern = s[:10]  #  

    start = time.perf_counter()
    compress_string(s)
    compress_times.append(time.perf_counter() - start)

    start = time.perf_counter()
    naive_search(s, pattern)
    naive_times.append(time.perf_counter() - start)

    start = time.perf_counter()
    z_function(s)
    zfunc_times.append(time.perf_counter() - start)

#  
plt.figure(figsize=(10, 6))
plt.plot(sizes, compress_times, label="  (RLE)", marker='o')
plt.plot(sizes, naive_times, label=" ", marker='o')
plt.plot(sizes, zfunc_times, label="Z-", marker='o')
plt.xlabel(" ")
plt.ylabel("  ()")
plt.title("     ")
plt.legend()
plt.grid(True)
plt.tight_layout()
